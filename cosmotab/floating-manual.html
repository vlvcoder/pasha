🎨 Как создать эффект парения

/* Базовая формула */
box-shadow: [смещение_X] [смещение_Y] [размытие] [растяжение] [цвет];

/* Для эффекта парения используйте: */
.element {
  box-shadow: 0 20px 25px -5px rgba(0,0,0,0.5);
  transition: all 0.3s ease;
}

/* При наведении усиливайте тень */
.element:hover {
  box-shadow: 0 30px 35px -5px rgba(0,0,0,0.6);
  transform: translateY(-5px);
}
📏 смещение_Y = 20-30px
💫 размытие = 25-40px
📐 растяжение = -5px (важно!)
⬆️ translateY при hover
